sub banking_status {
	my ($self, $args) = @_;

	my $item = {};
	$zeny->{zeny} = $args->{zeny};
	$reason->{reason} = $args->{reason};

	my $msg;
	$msg .= center(" " . T("Bank_info") . " ", 79, '-') . "\n";
	$msg .= swrite(TF("Zeny: \@%s, ('<'x39), ('<'x24)),
			[bytesToString($args->{zeny}))]);
	$msg .= TF("Reason: %s\n", bytesToString($args->{reason}));
	$msg .= ("%s\n", ('-'x79));
	$msg .= sprintf("%s\n", ('-'x79));
	message($msg, "info");
}

sub cmdBank {
    my (undef, $args) = @_;
    my ($arg1, $arg2) = parseArgs($args, 2);

        error TF("You must be logged in the game to use this command\n");
    }
    if ($arg1 eq "check") {
         $messageSender->banking_check();
        return;

        error TF("Unknown Bank command\n");
    }
}
